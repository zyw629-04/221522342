def preprocess_industry_data(industry_data, scale_features=True):
    """
    行业数据预处理
    
    参数:
    industry_data: 行业数据DataFrame
    scale_features: 是否对数值特征进行标准化
    
    返回:
    预处理后的DataFrame和标准化器(如果scale_features为True)
    """
    # 数据副本
    df = industry_data.copy()
    
    # 1. 缺失值处理
    numeric_cols = df.select_dtypes(include=[np.number]).columns
    imputer = SimpleImputer(strategy='median')
    df[numeric_cols] = imputer.fit_transform(df[numeric_cols])
    
    # 2. 异常值处理 - 基于Z-score
    z_threshold = 3
    for col in numeric_cols:
        z_scores = np.abs((df[col] - df[col].mean()) / df[col].std())
        df = df[z_scores < z_threshold]
    
    # 3. 特征标准化
    scaler = None
    if scale_features:
        scaler = StandardScaler()
        df[numeric_cols] = scaler.fit_transform(df[numeric_cols])
    
    return df, scaler

def preprocess_skill_demand_data(skill_data, fill_method='ffill', scale_method='minmax'):
    """
    技能需求时间序列数据预处理
    
    参数:
    skill_data: 技能需求数据DataFrame
    fill_method: 缺失值填充方法 ('ffill', 'bfill', 'median'等)
    scale_method: 标准化方法 ('minmax'或'standard')
    
    返回:
    预处理后的DataFrame和标准化器
    """
    # 数据副本
    df = skill_data.copy()
    
    # 1. 缺失值填充
    if fill_method == 'median':
        df = df.fillna(df.median())
    else:
        df = df.fillna(method=fill_method)
    
    # 2. 平稳性检查与处理 (简化版)
    # 这里仅打印提示，实际项目中可添加ADF检验和差分处理
    print(u"技能需求数据平稳性检查: 建议对非平稳序列进行差分处理")
    
    # 3. 数据标准化
    scaler = None
    if scale_method == 'minmax':
        scaler = MinMaxScaler()
    elif scale_method == 'standard':
        scaler = StandardScaler()
    
    if scaler:
        df = pd.DataFrame(scaler.fit_transform(df), columns=df.columns, index=df.index)
    
    return df, scaler

def preprocess_policy_data(policy_data, encode_categorical=True):
    """
    政策影响数据预处理
    
    参数:
    policy_data: 政策数据DataFrame
    encode_categorical: 是否对分类特征进行编码
    
    返回:
    预处理后的DataFrame和编码器(如果encode_categorical为True)
    """
    df = policy_data.copy()
    
    # 1. 缺失值处理
    imputer = SimpleImputer(strategy='mean')
    df = pd.DataFrame(imputer.fit_transform(df), columns=df.columns, index=df.index)
    
    # 2. 分类特征编码 (假设无分类特征，此处为示例)
    encoder = None
    if encode_categorical:
        categorical_cols = df.select_dtypes(include=['object']).columns
        if len(categorical_cols) > 0:
            encoder = OneHotEncoder(sparse=False, handle_unknown='ignore')
            encoded = encoder.fit_transform(df[categorical_cols])
            encoded_df = pd.DataFrame(encoded, columns=encoder.get_feature_names(categorical_cols))
            df = pd.concat([df.drop(categorical_cols, axis=1), encoded_df], axis=1)
    
    return df, encoder

def preprocess_supply_demand_data(supply_demand, scale_ratio=True):
    """
    供需数据分析预处理
    
    参数:
    supply_demand: 供需数据DataFrame
    scale_ratio: 是否对供需比进行标准化
    
    返回:
    预处理后的DataFrame
    """
    df = supply_demand.copy()
    
    # 1. 缺失值处理
    numeric_cols = df.select_dtypes(include=[np.number]).columns
    df[numeric_cols] = df[numeric_cols].fillna(df[numeric_cols].median())
    
    # 2. 供需比标准化
    if scale_ratio and '供需比' in df.columns:
        scaler = MinMaxScaler()
        df['供需比'] = scaler.fit_transform(df[['供需比']])
    
    # 3. 特征衍生
    df['供需状态'] = df['供需比'].apply(lambda x: u'供不应求' if x < 1 else u'供过于求')
    
    return df

def preprocess_all_data(industry_data, skill_data, policy_data, supply_demand, 
                       scale_industry=True, scale_skill='minmax', 
                       encode_policy=True, scale_supply_ratio=True):
    """
    批量预处理所有类型数据
    
    参数:
    返回:
    预处理后的数据字典
    """
    print(u"开始数据预处理...")
    
    # 行业数据预处理
    print(u"处理行业数据...")
    industry_processed, industry_scaler = preprocess_industry_data(industry_data, scale_industry)
    
    # 技能需求数据预处理
    print(u"处理技能需求数据...")
    skill_processed, skill_scaler = preprocess_skill_demand_data(skill_data, scale_method=scale_skill)
    
    # 政策数据预处理
    print(u"处理政策数据...")
    policy_processed, policy_encoder = preprocess_policy_data(policy_data, encode_policy)
    
    # 供需数据预处理
    print(u"处理供需数据...")
    supply_processed = preprocess_supply_demand_data(supply_demand, scale_supply_ratio)
    
    print(u"数据预处理完成!")
    
    return {
        'industry': industry_processed,
        'skill': skill_processed,
        'policy': policy_processed,
        'supply_demand': supply_processed,
        'scalers': {
            'industry': industry_scaler,
            'skill': skill_scaler,
            'policy': policy_encoder
        }
    }

def save_preprocessed_data(processed_data, save_dir="预处理后人才市场数据"):
    """
    保存预处理后的数据
    
    参数:
    processed_data: 预处理后的数据字典
    save_dir: 保存目录
    """
    try:
        # 创建保存目录
        if not os.path.exists(save_dir):
            os.makedirs(save_dir)
        
        # 保存行业数据
        processed_data['industry'].to_csv(os.path.join(save_dir, "行业数据_预处理.csv"), index=False, encoding='utf-8')
        
        # 保存技能需求数据
        processed_data['skill'].to_csv(os.path.join(save_dir, "技能需求数据_预处理.csv"), index=True, encoding='utf-8')
        
        # 保存政策数据
        processed_data['policy'].to_csv(os.path.join(save_dir, "政策影响数据_预处理.csv"), index=False, encoding='utf-8')
        
        # 保存供需数据
        processed_data['supply_demand'].to_csv(os.path.join(save_dir, "供需数据_预处理.csv"), index=False, encoding='utf-8')
        
        print(u"预处理后的数据已成功保存到 {} 目录下".format(os.path.abspath(save_dir)))
        
    except Exception as e:
        print(u"保存预处理数据时出错: {}".format(e))

def main(industry_file, skill_file, policy_file, supply_file, save_dir="预处理后人才市场数据"):
    """
    主函数 - 数据预处理流程
    
    参数:
    industry_file: 行业数据文件路径
    skill_file: 技能需求数据文件路径
    policy_file: 政策数据文件路径
    supply_file: 供需数据文件路径
    save_dir: 保存目录
    """
    # 读取数据
    print(u"读取原始数据...")
    industry_data = pd.read_csv(industry_file, encoding='utf-8')
    skill_data = pd.read_csv(skill_file, encoding='utf-8', index_col=0)
    policy_data = pd.read_csv(policy_file, encoding='utf-8')
    supply_demand = pd.read_csv(supply_file, encoding='utf-8')
    
    # 数据预处理
    processed_data = preprocess_all_data(
        industry_data, skill_data, policy_data, supply_demand,
        scale_industry=True,
        scale_skill='minmax',
        encode_policy=True,
        scale_supply_ratio=True
    )
    
    # 保存预处理后的数据
    save_preprocessed_data(processed_data, save_dir)

def radar_chart(data, title, ax):
    categories = list(data.columns)
    N = len(categories)
    
    # 闭合图形
    values = data.values.flatten().tolist()
    values += values[:1]
    
    # 角度计算
    angles = np.linspace(0, 2*np.pi, N, endpoint=False).tolist()
    angles += angles[:1]
    
    # 绘制雷达图
    ax.polar(angles, values, 'o-', linewidth=2)
    ax.fill(angles, values, alpha=0.25)
    ax.set_thetagrids(np.degrees(angles[:-1]), categories)
    ax.set_title(title, fontsize=12, y=1.1)
    ax.grid(True)

# -*- coding: utf-8 -*-
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.arima_model import ARIMA
from sklearn.metrics import mean_squared_error

# 配置中文显示
plt.rcParams["font.family"] = ["SimHei"]
plt.rcParams["axes.unicode_minus"] = False

# 生成模拟数据
np.random.seed(42)
n = 60
trend = np.linspace(100, 300, n)
seasonal = 50 * np.sin(np.linspace(0, 10*np.pi, n))
noise = np.random.normal(0, 20, n)
demand = trend + seasonal + noise

# 创建数据框
dates = pd.date_range('2018-01', periods=n, freq='MS')
df = pd.DataFrame({'需求': demand}, index=dates)

# 数据可视化
plt.figure(figsize=(10, 4))
plt.plot(df)
plt.title('月度人才需求时间序列')
plt.grid(True, alpha=0.3)
plt.show()

# 划分训练测试集
train, test = df.iloc[:-12], df.iloc[-12:]

# ARIMA模型
model = ARIMA(train, order=(5, 1, 0)).fit(disp=0)
pred = model.predict(start=len(train), end=len(df)-1)

# 评估
mse = mean_squared_error(test, pred)
rmse = np.sqrt(mse)
print("MSE: %.2f, RMSE: %.2f" % (mse, rmse))

# 可视化预测
plt.figure(figsize=(10, 4))
plt.plot(train, label='训练数据')
plt.plot(test, label='实际值', alpha=0.7)
plt.plot(pred, label='ARIMA预测', color='red', linestyle='--')
plt.title('人才需求预测')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()

# 未来预测
forecast = model.forecast(steps=12)[0]
future = pd.Series(forecast, index=pd.date_range(df.index[-1], periods=13, freq='MS')[1:])

# 未来可视化
plt.figure(figsize=(10, 4))
plt.plot(df, label='历史数据')
plt.plot(future, label='未来12月预测', color='red', linestyle='--')
plt.title('人才需求未来预测')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()

try:
    from statsmodels.tsa.statespace.sarimax import SARIMAX
    smodel = SARIMAX(train, order=(1,1,1), seasonal_order=(1,1,1,12)).fit(disp=0)
    spred = smodel.predict(len(train), len(df)-1)
    smse = mean_squared_error(test, spred)
    print("\nSARIMA MSE: %.2f" % smse)
    
    plt.figure(figsize=(10, 4))
